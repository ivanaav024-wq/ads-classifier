SQ Cleanup · One-time setup

Setup instructions

This sets up the automated ads classifier. Once done, the app reads Excel files from your Drive every Monday at 8am, classifies the search terms using AI, writes the results directly to Google Sheets, and emails you a summary — without anyone needing to do anything.

This only needs to be done once. It takes about 20 minutes.

What you will do in these 7 steps

1
Open the sq-cleanup project in Google Cloud Console
2
Enable two APIs — Google Sheets and Gemini (the AI)
3
Create a free Gemini API key — what powers the AI classification
4
Configure the sign-in screen — so only you and your nephew can use the app
5
Create an OAuth Client ID — tells Google the website is allowed to show a sign-in button
6
Find your Automation folder ID from the Drive URL
7
Paste and deploy the backend script — this holds the keys and does all the work on Google's servers
1
Open the sq-cleanup project
Google Cloud Console · 1 minute
▾

2
Enable two APIs
Google Sheets API + Gemini API · 3 minutes
▾

You already enabled Google Drive API previously — that one is done. We need two more.

  • A

  • B
    Go back to the Library. Search for Generative Language API (this is Gemini, Google's AI). Click it, then click Enable.
✓ Google Drive API is already enabled — no need to touch it.
3

The free AI key · 2 minutes
▾

  • →
    Click + Create Credentials → API key. A key appears — it starts with AIza. Copy it and save it in a notes app.

Keep this key safe — you paste it into the script in Step 7. You do not send it to your nephew. It stays inside the script on Google's servers.

7
Deploy the backend script
Holds all keys, does all the work on Google's servers · 8 minutes
▾

This script runs entirely on Google's servers. Your nephew's browser never sees the Gemini key. The script reads the Excel files from Drive, calls the AI, writes results to Sheets, and sends the email summary. You paste it once and never touch it again.

  • A
    Open this in a new tab:
    https://script.google.com
    Click New project (top left).
  • B
    A code editor opens with some placeholder text already in it. Click inside it, press Ctrl+A (Cmd+A on Mac) to select everything, then press Delete. The editor should now be completely empty.
  • C
    Click the Copy code button below, then click inside the empty editor and press Ctrl+V (Cmd+V on Mac) to paste.
    // ================================================================ // SQ CLEANUP — ADS CLASSIFIER BACKEND // Google Apps Script // ================================================================ const CONFIG = { GEMINI_API_KEY: 'PASTE_YOUR_GEMINI_KEY_HERE', AUTOMATION_FOLDER_ID: 'PASTE_YOUR_FOLDER_ID_HERE', OUTPUT_SHEET_ID: '', OUTPUT_TAB_NAME: 'SQ Cleanup', ALLOWED_EMAILS: ['your@email.com', 'nephews@email.com'], }; const SYSTEM_PROMPT = `You are a Google Ads search term classifier for a digital marketing agency. Classify each search term as IRRELEVANT, COMPETITOR, or KEEP. IRRELEVANT: wrong intent, wrong audience, generic/unrelated — will never convert. COMPETITOR: clearly names or strongly implies a direct competitor brand. KEEP: relevant, branded, or has conversion potential — do NOT classify these. Be conservative. When in doubt, use KEEP. It is worse to remove a good term than keep a bad one. Respond ONLY with valid JSON, no extra text, no markdown.`; function doPost(e) { try { const body = JSON.parse(e.postData.contents); const userEmail = body.userEmail || ''; if (!CONFIG.ALLOWED_EMAILS.includes(userEmail)) { return jsonResponse({ ok: false, error: 'Unauthorised: ' + userEmail }); } if (body.action === 'list_files') return jsonResponse(listFiles()); if (body.action === 'classify_file') return jsonResponse(classifyFile(body.fileId, body.company)); if (body.action === 'write_results') return jsonResponse(writeResults(body.results)); if (body.action === 'full_run') return jsonResponse(fullRun()); return jsonResponse({ ok: false, error: 'Unknown action' }); } catch (err) { return jsonResponse({ ok: false, error: err.message }); } } function doGet() { return jsonResponse({ ok: true, message: 'SQ Cleanup backend is running.' }); } function listFiles() { const folder = DriveApp.getFolderById(CONFIG.AUTOMATION_FOLDER_ID); const files = []; const it = folder.getFilesByType(MimeType.MICROSOFT_EXCEL); while (it.hasNext()) { const f = it.next(); files.push({ id: f.getId(), name: f.getName() }); } const it2 = folder.getFilesByType('application/vnd.ms-excel'); while (it2.hasNext()) { const f = it2.next(); if (!files.find(x => x.id === f.getId())) files.push({ id: f.getId(), name: f.getName() }); } files.sort((a, b) => a.name.localeCompare(b.name)); return { ok: true, files }; } function classifyFile(fileId, company) { const terms = parseExcelFile(fileId); if (!terms.length) return { ok: true, company, irrelevant: [], competitors: [], totalTerms: 0 }; const irrelevant = [], competitors = []; const BATCH = 25; for (let i = 0; i < terms.length; i += BATCH) { const res = callGemini(terms.slice(i, i + BATCH), company); for (const r of res) { if (r.label === 'IRRELEVANT') irrelevant.push(r.term); else if (r.label === 'COMPETITOR') competitors.push(r.term); } Utilities.sleep(4000); } return { ok: true, company, irrelevant, competitors, totalTerms: terms.length }; } function fullRun() { const { files } = listFiles(); const allResults = {}; for (const file of files) { const company = file.name.replace(/\.(xlsx?|csv)$/i, '').replace(/[_-]/g, ' ').trim(); try { const r = classifyFile(file.id, company); allResults[company] = { irrelevant: r.irrelevant, competitors: r.competitors }; } catch (err) { allResults[company] = { irrelevant: [], competitors: [], error: err.message }; } } writeResults(allResults); return { ok: true, results: allResults }; } function writeResults(results) { const companies = Object.keys(results); if (!companies.length) return { ok: false, error: 'No results' }; let sheetId = CONFIG.OUTPUT_SHEET_ID || PropertiesService.getScriptProperties().getProperty('OUTPUT_SHEET_ID'); if (!sheetId) { const weekStr = Utilities.formatDate(new Date(), 'America/New_York', 'yyyy-MM-dd'); const ss = SpreadsheetApp.create('SQ Cleanup — ' + weekStr); sheetId = ss.getId(); PropertiesService.getScriptProperties().setProperty('OUTPUT_SHEET_ID', sheetId); } const ss = SpreadsheetApp.openById(sheetId); let sheet = ss.getSheetByName(CONFIG.OUTPUT_TAB_NAME); if (!sheet) sheet = ss.insertSheet(CONFIG.OUTPUT_TAB_NAME); sheet.clearContents(); sheet.clearFormats(); const maxRows = Math.max(1, ...companies.map(c => Math.max((results[c].irrelevant||[]).length, (results[c].competitors||[]).length))); const row1 = ['#'], row2 = ['#']; for (const co of companies) { row1.push(co, ''); row2.push('Irrelevant', 'Competitor'); } const data = [row1, row2]; for (let r = 0; r < maxRows; r++) { const row = [r + 1]; for (const co of companies) { row.push((results[co].irrelevant||[])[r]||'', (results[co].competitors||[])[r]||''); } data.push(row); } const lastCol = 1 + companies.length * 2; sheet.getRange(1, 1, data.length, data[0].length).setValues(data); sheet.setFrozenRows(2); sheet.setFrozenColumns(1); sheet.getRange(1, 1, 1, lastCol).setBackground('#1a237e').setFontColor('#fff').setFontWeight('bold'); for (let c = 0; c < companies.length; c++) { sheet.getRange(2, 2+c*2).setBackground('#1b5e20').setFontColor('#fff').setFontWeight('bold'); sheet.getRange(2, 3+c*2).setBackground('#b71c1c').setFontColor('#fff').setFontWeight('bold'); } sheet.autoResizeColumns(1, lastCol); const sheetUrl = 'https://docs.google.com/spreadsheets/d/' + sheetId; return { ok: true, sheetId, sheetUrl, rowsWritten: data.length }; } function parseExcelFile(fileId) { const tempId = Drive.Files.copy({ title: '_temp' }, fileId, { convert: true }).id; const tempSS = SpreadsheetApp.openById(tempId); try { const data = tempSS.getSheets()[0].getDataRange().getValues(); if (data.length < 2) return []; const headers = data[0].map(h => String(h).toLowerCase()); let col = headers.findIndex(h => h.includes('search term') || h.includes('keyword') || h.includes('query')); if (col === -1) col = 0; return data.slice(1).map(r => String(r[col]||'').trim()).filter(t => t.length > 0); } finally { DriveApp.getFileById(tempSS.getId()).setTrashed(true); } } function callGemini(terms, company) { const prompt = SYSTEM_PROMPT + '\n\nCompany: ' + company + '\n\nSearch terms:\n' + terms.map((t,i) => (i+1)+'. '+t).join('\n') + '\n\nJSON only: {"results":[{"term":"...","label":"IRRELEVANT"|"COMPETITOR"|"KEEP"}]}'; const r = UrlFetchApp.fetch( 'https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key='+CONFIG.GEMINI_API_KEY, { method:'post', contentType:'application/json', muteHttpExceptions:true, payload: JSON.stringify({contents:[{parts:[{text:prompt}]}],generationConfig:{temperature:0.1,maxOutputTokens:2048}}) } ); if (r.getResponseCode() !== 200) throw new Error('Gemini error: ' + r.getContentText()); try { const text = JSON.parse(r.getContentText()).candidates[0].content.parts[0].text; return JSON.parse(text.replace(/```json|```/g,'')).results || []; } catch(e) { return []; } } function scheduledWeeklyRun() { try { const result = fullRun(); const companies = Object.keys(result.results||{}); const totalIrrel = companies.reduce((s,c)=>s+(result.results[c].irrelevant?.length||0),0); const totalComp = companies.reduce((s,c)=>s+(result.results[c].competitors?.length||0),0); MailApp.sendEmail({ to: CONFIG.ALLOWED_EMAILS.join(','), subject: '✅ SQ Cleanup — Weekly Run Complete', body: 'Classification finished.\n\nCompanies: '+companies.length+ '\nIrrelevant terms: '+totalIrrel+'\nCompetitor terms: '+totalComp+ '\n\nOpen results:\n'+(result.sheetUrl||'') }); } catch(err) { MailApp.sendEmail({to:CONFIG.ALLOWED_EMAILS[0],subject:'❌ SQ Cleanup — Run Failed',body:'Error: '+err.message}); } } function jsonResponse(data) { return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON); }
